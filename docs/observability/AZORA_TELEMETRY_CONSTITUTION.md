struct TelemetryResourceAllocation {

    fn calculate_effective_quota(&self) -> ResourceQuota {
        let base = self.base_allocation.amount;
        let pivc_bonus = self.pivc_bonus.amount;
        let sovereignty_boost = base * self.sovereignty_multiplier;

            amount: (base + pivc_bonus + sovereignty_boost) * constitutional_boost,
            unit: self.base_allocation.unit,
        }
    }
}
